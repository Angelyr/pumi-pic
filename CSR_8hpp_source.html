186  template <typename FunctionType>
187  void CSR<DataTypes, MemSpace>::parallel_for(FunctionType& fn, std::string name) {
188  if (nPtcls() == 0)
189  return;
190  FunctionType* fn_d = gpuMemcpy(fn);
191  const lid_t league_size = num_elems;
192  const lid_t team_size = policy.team_size();
193  const PolicyType policy(league_size, team_size);
194  auto offsets_cpy = offsets;
195  lid_t num_ptcls_cpy = num_ptcls;
196  Kokkos::parallel_for(name, policy,
197  KOKKOS_LAMBDA(const typename PolicyType::member_type& thread) {
198  const lid_t elm = thread.league_rank();
199  const lid_t start = offsets_cpy(elm);
200  const lid_t end = offsets_cpy(elm+1);
201  const lid_t numPtcls = end-start;
202  Kokkos::parallel_for(Kokkos::TeamThreadRange(thread, numPtcls), [=] (lid_t& j) {
203  const lid_t particle_id = start+j;
204  bool mask = true;
205  if (particle_id > num_ptcls_cpy)
206  mask = false;
207  (*fn_d)(elm, particle_id, mask);
208  });
209  });
210 #ifdef PP_USE_GPU
211  gpuFree(fn_d);
212 #endif
213  }

271  if (std::is_same<memory_space, typename MSpace::memory_space>::value) {
272  pPrintError( "[ERROR] Copy to same memory space not supported\n");
273  exit(EXIT_FAILURE);
274  }
275  Mirror<MSpace>* mirror_copy = new CSR<DataTypes, MSpace>();
276  //Call Particle structures copy
277  mirror_copy->copy(this);
278  //Copy constants
279  mirror_copy->swap_capacity_ = swap_capacity_;
280  mirror_copy->always_realloc = always_realloc;
281  mirror_copy->minimize_size = minimize_size;
282  mirror_copy->padding_amount = padding_amount;
283 
284  //Create the swap space
285  mirror_copy->ptcl_data_swap = createMemberViews<DataTypes, memory_space>(swap_capacity_);
286  //Deep copy each view
287  mirror_copy->offsets = typename Mirror<MSpace>::kkLidView("mirror offsets", offsets.size());
288  Kokkos::deep_copy(mirror_copy->offsets, offsets);
289  mirror_copy->element_to_gid = typename Mirror<MSpace>::kkGidView("mirror element_to_gid",
290  element_to_gid.size());
291  Kokkos::deep_copy(mirror_copy->element_to_gid, element_to_gid);
292  //Deep copy the gid mapping
293  mirror_copy->element_gid_to_lid.create_copy_view(element_gid_to_lid);
294  return mirror_copy;
