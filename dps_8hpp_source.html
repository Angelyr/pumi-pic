123  template<class DataTypes, typename MemSpace>
124  void DPS<DataTypes, MemSpace>::construct(kkLidView ptcls_per_elem,
125  kkGidView element_gids,
126  kkLidView particle_elements,
127  MTVs particle_info,
128  MPI_Comm mpi_comm) {
129  Kokkos::Profiling::pushRegion("dps_construction");
130  int comm_rank;
131  MPI_Comm_rank(mpi_comm, &comm_rank);
132  if(!comm_rank)
133  pPrintInfo( "building DPS\n");
134 
135  // calculate num_soa_ from number of particles + extra padding
136  num_soa_ = Kokkos::ceil(Kokkos::ceil(double(num_ptcls)/AoSoA_t::vector_length)*(1+extra_padding));
137  // calculate capacity_ from num_soa_ and max size of an SoA
138  capacity_ = num_soa_*AoSoA_t::vector_length;
139  // initialize appropriately-sized AoSoA
140  aosoa_ = makeAoSoA(capacity_, num_soa_);
141  // set active mask
142  setNewActive(num_ptcls);
143  // get global ids
144  if (element_gids.size() > 0)
145  createGlobalMapping(element_gids, element_to_gid, element_gid_to_lid);
146  // populate AoSoA with input data if given
147  if (particle_elements.size() > 0 && particle_info != NULL) {
148  if(!comm_rank) pPrintInfo( "initializing DPS data\n");
149  fillAoSoA(particle_elements, particle_info, parentElms_); // fill aosoa with data
150  }
151  else
152  setParentElms(ptcls_per_elem, parentElms_);
153 
154  Kokkos::Profiling::popRegion();
155  }

244  if (std::is_same<memory_space, typename MSpace::memory_space>::value) {

246  exit(EXIT_FAILURE);
247  }

249  //Call Particle structures copy
250  mirror_copy->copy(this);
251  //Copy constants
252  mirror_copy->num_soa_ = num_soa_;
253  mirror_copy->extra_padding = extra_padding;
254 
255  //Copy AoSoA
256  mirror_copy->aosoa_ = new typename DPS<DataTypes, MSpace>::AoSoA_t(std::string(aosoa_->label()).append("_mirror"), aosoa_->size());
257  Cabana::deep_copy(*(mirror_copy->aosoa_), *aosoa_);
258 
259  //Deep copy each view
260  mirror_copy->parentElms_ = typename Mirror<MSpace>::kkLidView("mirror parentElms_",
261  parentElms_.size());
262  Kokkos::deep_copy(mirror_copy->parentElms_, parentElms_);
263  mirror_copy->element_to_gid = typename Mirror<MSpace>::kkGidView("mirror element_to_gid",
264  element_to_gid.size());
265  Kokkos::deep_copy(mirror_copy->element_to_gid, element_to_gid);
266  //Deep copy the gid mapping
267  mirror_copy->element_gid_to_lid.create_copy_view(element_gid_to_lid);
268  return mirror_copy;

273  // Sum number of empty cells
274  auto mask = Cabana::slice<DPS_DT::size-1>(*aosoa_);
275  kkLidView padded_cells("num_padded_cells",1);
276  Kokkos::parallel_for("count_padding", capacity_,
277  KOKKOS_LAMBDA(const lid_t ptcl_id) {
278  Kokkos::atomic_fetch_add(&padded_cells(0), !mask(ptcl_id));
279  });
280  lid_t num_padded = getLastValue(padded_cells);
281

283  MPI_Comm_rank(mpi_comm, &comm_rank);
284  char buffer[1000];

286  // Header
287  ptr += sprintf(ptr, "Metrics (Rank %d)\n", comm_rank);
288  // Sizes

290  num_elems, num_soa_, num_ptcls, capacity_);
291  // Padded Cells
292  ptr += sprintf(ptr, "Padded Cells <Tot %%> %d %.3f%%\n", num_padded,
293  num_padded * 100.0 / capacity_);
